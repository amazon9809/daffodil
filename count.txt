<?php
// 카운터 파일 경로
$file = 'count.txt';

// 파일이 없으면 생성
if (!file_exists($file)) {
    file_put_contents($file, 0);
}

// 파일 읽기
$count = file_get_contents($file);

// 페이지 새로고침 시 숫자가 올라가지 않게 하려면 세션 처리 필요
session_start();
if (!isset($_SESSION['visited'])) {
    $count++;
    file_put_contents($file, $count);
    $_SESSION['visited'] = true;
}

echo "현재까지 총 방문자 수: " . $count;
?>
